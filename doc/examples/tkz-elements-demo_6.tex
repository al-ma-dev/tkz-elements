\documentclass[margin = 12pt]{standalone}  
\usepackage[mini]{tkz-euclide}
\usepackage{tkz-elements}

\begin{document} 
    
\directlua{
  init_elements()
  z.A = point(0, 0)
  z.B = point(1.5, -3)
  z.C = point(4, 1)
  T.ABC = triangle(z.A, z.B, z.C)
  C.ABC = T.ABC:circum_circle()
  z.w, z.t = C.ABC:get()
  z.P = C.ABC:point(-0.15)
  L.steiner = T.ABC:steiner_line(z.P)
  z.sa, z.sb = L.steiner:get()
  z.H = T.ABC.orthocenter
  CO.PA = conic:new(z.P, L.steiner, 1)
  PA.curve = CO.PA:points(-5, 5, 50)
  z.M = intersection(CO.PA, T.ABC.ca)
}
    
\begin{tikzpicture}
  \tkzGetNodes
  \tkzDrawCoordinates[smooth,red](PA.curve)
  \tkzDrawLines[add = 1 and 1,blue](A,B B,C A,C)
  \tkzDrawLines[cyan,add = .4 and .6](sa,sb)
  \tkzDrawCircle[orange](w,t)
  \tkzDrawPoints (A,B,C,P,H,M)
  \tkzLabelPoints(A,B,C,P,H,M)
\end{tikzpicture}
\end{document}
